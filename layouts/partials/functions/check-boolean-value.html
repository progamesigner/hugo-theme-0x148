{{- $item := 0 | index . -}} {{- $enabledItems := 1 | index . | default slice -}} {{- $disabledItems := 2 | index . | default slice -}} {{- $name := "name" | lower | index $item -}} {{- $enabled := "enable" | lower | index $item -}} {{- if $enabled | eq false -}} {{- $enabled = $name | in $enabledItems -}} {{- else -}} {{- $enabled = $name | in $disabledItems | not -}} {{- end -}} {{- return $enabled -}}
